
#Initalize Git respo.
> git init

#check status
>git status

#stageing / unstageing
>git add ______
>git add "*.txt"     #adds all .txt files
>git reset ______  #unstageing



#commiting 
>git commit -m "message"

#git log
>git log

#make/add remote respo
>git remote add origin https://

#push to remote respo
>git push -u origin master        #-u = remember the parameters, origin = added respo, master = loacal branch

#pull: get changes from github
>git pull origin master

#differences
>git diff HEAD       #HEAD = most recent changes

#looking at staged differences : staged files are files ready to be commited
>git add  _files
>git diff --staged

#change files back to before commmit
>git checkout --files

#create branch / copy of master?
>git branch _branch_name

#delete branch
>git branch -d _branch_name

#change branch / master
>git checout _branch_name
>git checkout master

#remove all
>git rm ____     #use wildcard "*.txt" if needed


#merge branches into master
>git checkout master
>git merge _branch_name
